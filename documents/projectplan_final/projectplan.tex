\begin{frame}
	\frametitle{Collision Avoidance}
	\begin{block}{Router instead of collision avoidance}
		\begin{itemize}
			\item calculates only save routes to target
			\item gets start point, target point and width of robot
			\item obstacles are circles
			\item considers only obstacles closer than 1 meter
			\item one route consists of one or more paths
			\item calculates every frame a new route
			\item takes the new route if its better
		\end{itemize}
	\end{block}
